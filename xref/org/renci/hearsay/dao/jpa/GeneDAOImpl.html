12  import org.renci.hearsay.dao.GeneDAO;
13  import org.renci.hearsay.dao.HearsayDAOException;
14  import org.renci.hearsay.dao.model.Gene;

16  import org.slf4j.Logger;

18  
19  public class GeneDAOImpl extends BaseEntityDAOImpl<Gene, Long> implements GeneDAO {
20  
21      private final Logger logger = LoggerFactory.getLogger(GeneDAOImpl.class);
22  
23      public GeneDAOImpl() {
24          super();
25      }
26  
27      @Override
28      public Class<Gene> getPersistentClass() {
29          return Gene.class;
30      }
31  
32      @Override
33      public List<Gene> findAll() throws HearsayDAOException {
34          logger.debug("ENTERING findAll()");
35          TypedQuery<Gene> query = getEntityManager().createNamedQuery("Gene.findAll", Gene.class);
36          List<Gene> ret = query.getResultList();
37          return ret;
38      }
39  
40      @Override
41      public List<Gene> findByName(String name) throws HearsayDAOException {
42          logger.debug("ENTERING findByName(Gene)");
43          CriteriaBuilder critBuilder = getEntityManager().getCriteriaBuilder();
44          CriteriaQuery<Gene> crit = critBuilder.createQuery(getPersistentClass());
45          Root<Gene> fromGene = crit.from(Gene.class);
46          List<Predicate> predicates = new ArrayList<Predicate>();
47          predicates.add(critBuilder.like(fromGene.get(Gene_.name), name));
48          crit.where(predicates.toArray(new Predicate[predicates.size()]));
49          TypedQuery<Gene> query = getEntityManager().createQuery(crit);
50          List<Gene> ret = query.getResultList();
51          return ret;
52      }

54  }
